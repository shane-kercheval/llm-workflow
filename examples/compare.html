    if not re.match(r"[^@]+@[^@]+\.[^@]+", email):

    # Split the email address into local and domain parts
    local_part, domain_part = email.split('@')

    # Mask the local part, keeping the first and last character unmasked
    masked_local = local_part[0] + "*" * (len(local_part) - 2) + local_part[-1] if len(local_part) > 2 else local_part

    # Combine the masked local part with the domain part
    masked_email = masked_local + "@" + domain_part

    return masked_email

# Example usage:
email = "example@example.com"

This function first checks if the provided email address is valid using a regular expression. Then it splits the email into the local and domain parts. It masks the local part by replacing all but the first and last characters with asterisks. If the local part is only one or two characters long, it doesn't mask any characters. Finally, it combines the masked local part with the domain part to return the masked email address.
